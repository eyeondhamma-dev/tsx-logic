import React from 'react';

export const compositionConfig = {
  id: 'Scene1_RelaxWarning',
  durationInSeconds: 6,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Scene1_RelaxWarning: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();
  
  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Background gradient
  const gradientRotate = interpolate(frame, [0, 180], [0, 360], { extrapolateRight: 'wrap' });

  // Particle system
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 3) * 1920,
    y: seededRandom(i * 5) * 1080,
    delay: i * 3,
    size: 3 + seededRandom(i * 7) * 4,
  }));

  // Warning icon shake
  const shakeX = Math.sin(frame * 0.3) * 3;
  const shakeY = Math.cos(frame * 0.25) * 2;

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>
      {/* Animated Gradient */}
      <AbsoluteFill
        style={{
          background: `linear-gradient(${gradientRotate}deg, #2D2D2D 0%, #1C1C1C 50%, #2D2D2D 100%)`,
        }}
      />

      {/* Floating Shape Layers */}
      <AbsoluteFill>
        <div

            position: 'absolute',
            top: '10%',
            left: '15%',
            width: 420,
            height: 420,
            borderRadius: '50%',
            border: '2px solid #e9d700',
            opacity: 0.05,
            transform: `rotate(${frame * 0.3}deg) scale(${1 + Math.sin(frame * 0.02) * 0.1})`,
          }}

            position: 'absolute',
            bottom: '15%',
            right: '10%',
            width: 350,
            height: 350,
            background: '#e9d700',
            opacity: 0.04,
            transform: `rotate(${frame * -0.2}deg)`,
          }}
        />
      </AbsoluteFill>

      {/* Particles */}
      {particles.map((p, i) => {
        const particleY = interpolate(frame, [p.delay, p.delay + 120], [p.y, p.y - 200], {
          extrapolateLeft: 'clamp',
          extrapolateRight: 'clamp',
        });

          extrapolateLeft: 'clamp',
          extrapolateRight: 'clamp',
        });
        
        return (
          <div
            key={i}
            style={{
              position: 'absolute',
              left: p.x,
              top: particleY,
              width: p.size,
              height: p.size,
              borderRadius: '50%',
              backgroundColor: '#e9d700',
              opacity: particleOp,
            }}
          />
        );
      })}

      {/* Main Content */}
      <AbsoluteFill style={{ justifyContent: 'center', alignItems: 'center' }}>
        {/* Warning Icon */}
        <div
          style={{
            transform: `translate(${shakeX}px, ${shakeY - 180}px) scale(${sp(0)})`,
            marginBottom: 40,

        >
          <svg width="140" height="140" viewBox="0 0 140 140">

            color: '#F5F5F5',
            textAlign: 'center',
            maxWidth: 1400,

            transform: `translateY(${interpolate(sp(15), [0, 1], [60, 0])}) scale(${sp(15)})`,
            opacity: sp(15),

          <svg width="140" height="140" viewBox="0 0 140 140">
            <circle cx="70" cy="70" r="60" fill="#e9d700" opacity="0.2" />
            <circle cx="70" cy="70" r="50" fill="none" stroke="#e9d700" strokeWidth="4" />
            
            {/* Cursor/Click icon */}
            <path
              d="M 55 40 L 55 80 L 70 70 L 80 90 L 90 85 L 80 65 L 95 65 Z"
              fill="#F5F5F5"
              stroke="#e9d700"
              strokeWidth="2"
            />
          </svg>
        </div>

        {/* Success rays emanating */}
        {successRays.map((ray, i) => {
          const rayLength = interpolate(frame, [ray.delay, ray.delay + 40], [0, 250], {
            extrapolateLeft: 'clamp',
            extrapolateRight: 'clamp',
          });
          const rayOp = interpolate(frame, [ray.delay, ray.delay + 10, ray.delay + 35, ray.delay + 40], [0, 0.6, 0.4, 0], {
            extrapolateLeft: 'clamp',
            extrapolateRight: 'clamp',
          });
          
          const rayX2 = 960 + Math.cos((ray.angle * Math.PI) / 180) * rayLength;
          const rayY2 = 350 + Math.sin((ray.angle * Math.PI) / 180) * rayLength;
          
          return (
            <svg key={i} width="1920" height="1080" style={{ position: 'absolute', top: 0, left: 0, pointerEvents: 'none' }}>
              <line
                x1="960"
                y1="350"
                x2={rayX2}
                y2={rayY2}
                stroke="#e9d700"
                strokeWidth="4"
                opacity={rayOp}
                strokeLinecap="round"
              />
            </svg>
          );
        })}

        {/* Victory burst particles */}
        {victoryParticles.map((p, i) => {
          const particleDistance = interpolate(frame, [p.delay, p.delay + 70], [0, 320 * p.speed], {
            extrapolateLeft: 'clamp',
            extrapolateRight: 'clamp',
          });
          const particleOp = interpolate(frame, [p.delay, p.delay + 15, p.delay + 60, p.delay + 70], [0, 1, 0.6, 0], {
            extrapolateLeft: 'clamp',
            extrapolateRight: 'clamp',
          });
          
          const pX = 960 + Math.cos(p.angle) * particleDistance;
          const pY = 350 + Math.sin(p.angle) * particleDistance;
          
          return (
            <div
              key={i}
              style={{
                position: 'absolute',
                left: pX,
                top: pY,
                width: 12,
                height: 12,
                borderRadius: '50%',
                backgroundColor: i % 4 === 0 ? '#e9d700' : i % 4 === 1 ? '#FFC107' : i % 4 === 2 ? '#GOLD' : '#F5F5F5',
                opacity: particleOp,
                boxShadow: `0 0 15px ${i % 2 === 0 ? '#e9d700' : '#FFC107'}`,
              }}
            />
          );
        })}

        {/* Trophy/Crown icon */}
        <div
          style={{
            position: 'absolute',
            top: '18%',
            transform: `scale(${sp(140)}) translateY(${Math.sin(frame * 0.06) * 6}px)`,
            opacity: sp(140),
          }}
        >
          <svg width="160" height="160" viewBox="0 0 160 160">
            {/* Crown */}
            <path
              d="M 40 100 L 50 50 L 80 70 L 110 50 L 120 100 Z"
              fill="#e9d700"
              stroke="#FFC107"
              strokeWidth="4"
            />
            <circle cx="50" cy="50" r="8" fill="#e9d700" stroke="#FFC107" strokeWidth="2" />
            <circle cx="80" cy="70" r="10" fill="#FFC107" stroke="#e9d700" strokeWidth="2" />
            <circle cx="110" cy="50" r="8" fill="#e9d700" stroke="#FFC107" strokeWidth="2" />
            
            {/* Base */}
            <rect x="45" y="100" width="70" height="15" rx="3" fill="#e9d700" stroke="#FFC107" strokeWidth="3" />
            
            {/* Sparkles */}
            <circle cx="30" cy="60" r="3" fill="#F5F5F5" opacity="0.8" />
            <circle cx="130" cy="65" r="3" fill="#F5F5F5" opacity="0.8" />
            <circle cx="80" cy="35" r="4" fill="#F5F5F5" opacity="0.9" />
          </svg>
        </div>

        {/* Text Section */}
        <div
          style={{
            position: 'absolute',
            bottom: '28%',
            fontFamily: 'Noto Sans Devanagari, sans-serif',
            fontSize: 64,
            fontWeight: 700,
            color: '#F5F5F5',
            textAlign: 'center',
            maxWidth: 1600,
            lineHeight: 1.4,
            transform: `translateY(${interpolate(sp(80), [0, 1], [50, 0])}) scale(${sp(80)})`,
            opacity: sp(80),
          }}
        >
          जो आपकी पूरे साल की सैलरी से भी ज्यादा हो सकता है
        </div>
        
        <div
          style={{
            position: 'absolute',
            bottom: '18%',
            fontFamily: 'Noto Sans Devanagari, sans-serif',
            fontSize: 76,
            fontWeight: 800,
            color: '#e9d700',
            textAlign: 'center',
            transform: `translateY(${interpolate(sp(100), [0, 1], [50, 0])}) scale(${sp(100)})`,
            opacity: sp(100),
            textShadow: '0 0 45px rgba(233, 215, 0, 0.6)',
          }}
        >
          सिर्फ एक क्लिक में आपके खाते में
        </div>

        <div
          style={{
            position: 'absolute',
            bottom: '8%',
            fontFamily: 'Noto Sans Devanagari, sans-serif',
            fontSize: 68,
            fontWeight: 700,
            color: '#FFC107',
            textAlign: 'center',
            maxWidth: 1700,
            lineHeight: 1.3,
            transform: `translateY(${interpolate(sp(120), [0, 1], [50, 0])}) scale(${sp(120)})`,
            opacity: sp(120),
          }}
        >
          क्योंकि आपने उस पत्थर को तब तक धकेला<br />
          जब तक वो ऊपर नहीं चढ़ गया!
        </div>
      </AbsoluteFill>

      {/* Stick Figure - VICTORIOUS */}
      <div
        style={{
          position: 'absolute',
          bottom: 80,
          left: '8%',
          transform: `scale(${sp(160)}) translateY(${Math.sin(frame * 0.15) * 10}px) rotate(${Math.sin(frame * 0.08) * 6}deg)`,
          opacity: sp(160),
        }}
      >
        <svg width="140" height="180" viewBox="0 0 140 180">
          {/* Mega glow */}
          <circle cx="70" cy="32" r="50" fill="#FFC107" opacity="0.4" />
          <circle cx="70" cy="32" r="35" fill="#e9d700" opacity="0.25" />
          
          {/* Head */}
          <circle cx="70" cy="32" r="26" fill="none" stroke="#F5F5F5" strokeWidth="4.5" />
          
          {/* Glasses - glowing */}
          <circle cx="61" cy="30" r="7.5" fill="none" stroke="#FFC107" strokeWidth="3" />
          <circle cx="79" cy="30" r="7.5" fill="none" stroke="#FFC107" strokeWidth="3" />
          <line x1="68.5" y1="30" x2="71.5" y2="30" stroke="#FFC107" strokeWidth="3" />
          
          {/* Body */}
          <line x1="70" y1="58" x2="70" y2="115" stroke="#F5F5F5" strokeWidth="4.5" strokeLinecap="round" />
          
          {/* Arms - VICTORY POSE */}
          <line x1="70" y1="75" x2="32" y2="50" stroke="#F5F5F5" strokeWidth="4.5" strokeLinecap="round" />
          <line x1="70" y1="75" x2="108" y2="50" stroke="#F5F5F5" strokeWidth="4.5" strokeLinecap="round" />
          
          {/* Legs - jumping */}
          <line x1="70" y1="115" x2="50" y2="160" stroke="#F5F5F5" strokeWidth="4.5" strokeLinecap="round" />
          <line x1="70" y1="115" x2="90" y2="160" stroke="#F5F5F5" strokeWidth="4.5" strokeLinecap="round" />
          
          {/* Victory sparkles */}
          <circle cx="32" cy="50" r="5" fill="#e9d700" opacity="0.9" />
          <circle cx="108" cy="50" r="5" fill="#e9d700" opacity="0.9" />
        </svg>
      </div>

      {/* Watermark */}
      <div
        style={{
          position: 'absolute',
          bottom: 30,
          right: 45,
          fontFamily: 'Inter, sans-serif',
          fontSize: 22,
          fontWeight: 600,
          color: '#9E9E9E',
          opacity: 0.3,
        }}
      >
        ProBuffet HQ
      </div>

      {/* Vignette - softer for victory */}
      <AbsoluteFill
        style={{
          background: 'radial-gradient(circle, transparent 55%, rgba(0,0,0,0.5) 100%)',
          pointerEvents: 'none',
        }}
      />
    </AbsoluteFill>
  );
};
