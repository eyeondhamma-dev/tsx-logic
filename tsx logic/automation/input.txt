  id: 'Chapter8Scene1',
  durationInSeconds: 10,
  fps: 30,
  width: 1920,
  height: 1080,
};

  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);

    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Camera zoom in
  const cameraZoom = interpolate(frame, [0, durationInFrames], [1, 1.06], {
    extrapolateRight: 'clamp',
  });

  // Paper entrance
  const paperScale = sp(20, { damping: 10, stiffness: 80 });
  const paperRotate = interpolate(frame, [20, 50], [8, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Line draw
  const lineHeight = interpolate(frame, [60, 110], [0, 100], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Left label
  const leftLabelOp = sp(120);
  // Right label
  const rightLabelOp = sp(145);

  // Pen entrance
  const penX = interpolate(frame, [50, 110], [200, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const penOp = interpolate(frame, [50, 60], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Floating particles
  const particles = Array.from({ length: 30 }, (_, i) => {
    const seed = seededRandom(i + 1);
    const seed2 = seededRandom(i + 100);
    const x = seed * 1920;
    const y = seed2 * 1080;
    const size = 2 + seed * 4;
    const speed = 0.3 + seed2 * 0.7;
    const drift = Math.sin(frame * 0.02 * speed + i) * 30;
    const floatY = Math.cos(frame * 0.015 * speed + i * 2) * 20;
    return { x: x + drift, y: y + floatY, size, opacity: 0.15 + seed * 0.25 };
  });

  // Shape layers
  const shape1Rotate = frame * 0.15;
  const shape2Rotate = frame * -0.1;
  const shape1Scale = 1 + Math.sin(frame * 0.02) * 0.05;
  const shape2Scale = 1 + Math.cos(frame * 0.025) * 0.04;

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>
      {/* Camera container */}
      <AbsoluteFill style={{ transform: `scale(${cameraZoom})` }}>

        {/* Dynamic gradient background */}
        <AbsoluteFill>
          <div style={{
            position: 'absolute', width: '100%', height: '100%',
            background: `radial-gradient(ellipse at 30% 70%, #2a1a0a 0%, transparent 50%),
                         radial-gradient(ellipse at 70% 30%, #0a1a2a 0%, transparent 50%),
                         radial-gradient(ellipse at 50% 50%, #1C1C1C 0%, #111111 100%)`,
          }} />
        </AbsoluteFill>

        {/* Animated shape layers */}
        <AbsoluteFill>
          {/* Large triangle */}
          <svg style={{
            position: 'absolute', left: -80, top: 100,
            transform: `rotate(${shape1Rotate}deg) scale(${shape1Scale})`,
            opacity: 0.05,
          }} width="600" height="600" viewBox="0 0 600 600">
            <polygon points="300,50 550,500 50,500" fill="none" stroke="#FFC107" strokeWidth="2" />
          </svg>
          {/* Large circle */}
          <svg style={{
            position: 'absolute', right: -100, bottom: -50,
            transform: `rotate(${shape2Rotate}deg) scale(${shape2Scale})`,
            opacity: 0.04,
          }} width="700" height="700" viewBox="0 0 700 700">
            <circle cx="350" cy="350" r="320" fill="none" stroke="#F5F5F5" strokeWidth="1.5" />
          </svg>
          {/* Hexagon */}
          <svg style={{
            position: 'absolute', left: '50%', top: -120,
            transform: `rotate(${frame * 0.08}deg) scale(${1 + Math.sin(frame * 0.03) * 0.03})`,
            opacity: 0.035,
          }} width="500" height="500" viewBox="0 0 500 500">
            <polygon points="250,30 450,140 450,360 250,470 50,360 50,140" fill="none" stroke="#e9d700" strokeWidth="1.5" />
          </svg>
        </AbsoluteFill>

        {/* Motion texture — diagonal lines */}
        <AbsoluteFill style={{ opacity: 0.03 }}>
          <svg width="1920" height="1080" viewBox="0 0 1920 1080">
            {Array.from({ length: 20 }, (_, i) => (
              <line
                key={i}
                x1={i * 100 + frame * 0.3}
                y1={0}
                x2={i * 100 - 200 + frame * 0.3}
                y2={1080}
                stroke="#F5F5F5"
                strokeWidth="0.5"
              />
            ))}
          </svg>
        </AbsoluteFill>

        {/* Floating particles */}
        <AbsoluteFill>
          {particles.map((p, i) => (
            <div key={i} style={{
              position: 'absolute',
              left: p.x, top: p.y,
              width: p.size, height: p.size,
              borderRadius: '50%',
              backgroundColor: i % 3 === 0 ? '#FFC107' : '#F5F5F5',

            justifyContent: 'flex-start',
            paddingTop: 50,
            overflow: 'hidden',
          }}>
            {/* Paper subtle lines */}
            {Array.from({ length: 14 }, (_, i) => (
              <div key={i} style={{
                position: 'absolute',
                top: 60 + i * 35,
                left: 60, right: 60,
                height: 1,
                backgroundColor: '#d0d0d0',
                opacity: 0.4,
              }} />
            ))}

            {/* Title on paper */}
            <div style={{
              fontFamily: "'Noto Sans Devanagari', sans-serif",
              fontSize: 32,
              fontWeight: 700,
              color: '#2D2D2D',
              opacity: interpolate(frame, [30, 50], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }),
              marginBottom: 20,
              textAlign: 'center',
              zIndex: 2,

              ५ मिनट का काम

            {/* Vertical line in center */}

              position: 'absolute',
              left: '50%',
              top: 100,
              width: 3,
              height: `${lineHeight * 3.5}%`,
              backgroundColor: '#ff0000',
              transform: 'translateX(-50%)',
              borderRadius: 2,
              zIndex: 2,

            {/* Left label — "लिया" */}
            <div style={{
              position: 'absolute',
              left: 80,
              top: 110,
              fontFamily: "'Noto Sans Devanagari', sans-serif",
              fontSize: 38,
              fontWeight: 800,
              color: '#ff0000',
              opacity: leftLabelOp,
              transform: `translateY(${(1 - leftLabelOp) * 20}px)`,
              zIndex: 2,

              पकड़े हैं ↓

            {/* Right label — "दिया" */}
            <div style={{
              position: 'absolute',
              right: 80,

            {/* Left items */}
            {['जमीन', 'मकान', 'सोना'].map((item, i) => {
