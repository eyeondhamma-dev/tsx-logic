import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Ch8Scene3',

  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Ch8Scene3: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Split screen reveal
  const leftReveal = interpolate(frame, [0, 30], [0, 1], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });
  const rightReveal = interpolate(frame, [20, 50], [0, 1], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  // Curve path draw progress
  const curveProgress = interpolate(frame, [40, 130], [0, 1], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Generate exponential curve points
  const generateCurveD = (progress: number) => {
    const points: string[] = [];
    const total = 80;
    const steps = Math.floor(total * progress);
    for (let i = 0; i <= steps; i++) {
      const t = i / total;
      const x = 80 + t * 560;
      const y = 380 - Math.pow(t, 1.8) * 300;
      points.push(`${i === 0 ? 'M' : 'L'}${x},${y}`);
    }
    return points.join(' ');
  };

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>

      {/* Split background */}
      <AbsoluteFill>
        {/* Left — Finance */}
        <div style={{
          position: 'absolute', left: 0, top: 0, width: 960, height: 1080,
          background: 'linear-gradient(160deg, #1a2a00 0%, #1C1C1C 100%)',
          clipPath: `inset(0 ${100 - leftReveal * 100}% 0 0)`,
        }} />
        {/* Right — Life */}
        <div style={{
          position: 'absolute', left: 960, top: 0, width: 960, height: 1080,
          background: 'linear-gradient(160deg, #1C1C1C 0%, #001a2a 100%)',
          clipPath: `inset(0 0 0 ${100 - rightReveal * 100}%)`,
        }} />
        {/* Center divider */}
        <div style={{
          position: 'absolute', left: 958, top: 0, width: 4, height: 1080,
          background: 'linear-gradient(180deg, transparent, #e9d700, transparent)',
          opacity: 0.6,
        }} />
      </AbsoluteFill>

      {/* Geometric shapes */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ position: 'absolute' }}>
          <circle cx="250" cy="900" r="300" fill="none" stroke="#e9d700" strokeWidth="1" opacity="0.05" />
          <circle cx="1700" cy="180" r="280" fill="none" stroke="#FFC107" strokeWidth="1" opacity="0.04" />
        </svg>
      </AbsoluteFill>

      {/* Left panel — Finance chart */}
      <AbsoluteFill style={{ opacity: leftReveal }}>
        <div style={{ position: 'absolute', left: 80, top: 80, fontFamily: 'Inter', fontSize: 40, fontWeight: 800, color: '#e9d700', letterSpacing: 3 }}>
          FINANCE
        </div>
        <svg width="960" height="1080" style={{ position: 'absolute' }}>
          {/* Chart axes */}
          <line x1="80" y1="400" x2="660" y2="400" stroke="#9E9E9E" strokeWidth="1.5" opacity="0.4" />
          <line x1="80" y1="80" x2="80" y2="400" stroke="#9E9E9E" strokeWidth="1.5" opacity="0.4" />

          {/* Y axis labels */}
          <text x="50" y="405" fill="#9E9E9E" fontSize="22" textAnchor="end" fontFamily="JetBrains Mono">₹1</text>
          <text x="50" y="280" fill="#9E9E9E" fontSize="22" textAnchor="end" fontFamily="JetBrains Mono">₹10</text>
          <text x="50" y="160" fill="#9E9E9E" fontSize="22" textAnchor="end" fontFamily="JetBrains Mono">₹100</text>

          {/* Compound curve */}
          <path d={generateCurveD(curveProgress)} fill="none" stroke="#e9d700" strokeWidth="4" strokeLinecap="round" />

          {/* Linear comparison */}
          <line x1="80" y1="400" x2={80 + 560 * curveProgress} y2={400 - 300 * curveProgress}
            stroke="#9E9E9E" strokeWidth="2" strokeDasharray="10,8" opacity="0.4" />

            <circle key={i} cx="960" cy="420" r={r}

                <rect key={i}
                  x={x - size / 2} y={y - size / 2}

                  fill={i % 3 === 0 ? '#e9d700' : i % 3 === 1 ? '#F5F5F5' : '#ff0000'}

        <svg width="1920" height="1080" style={{ position: 'absolute' }}>
          <polygon

            fill="none" stroke="#e9d700" strokeWidth="2" opacity="0.06"
            transform={`rotate(${frame * 0.02}, 960, 100)`}

          display: 'flex', flexDirection: 'column', alignItems: 'center',

          <div style={{
            fontFamily: 'JetBrains Mono',
            fontSize: 36,
            color: '#9E9E9E',
            letterSpacing: 12,
            fontWeight: 600,
            marginBottom: 4,

            PRO

          <div style={{
            fontFamily: 'Inter',

            BUFFET
          </div>

          <div style={{
            fontFamily: 'Inter',
            fontSize: 80,
            fontWeight: 900,
            color: '#F5F5F5',
            letterSpacing: 20,
          }}>
            HQ
          </div>

          {/* Tagline */}
          <div style={{
            fontFamily: 'Noto Sans Devanagari',
            fontSize: 38,
            color: '#9E9E9E',
            fontWeight: 400,
            marginTop: 12,
            letterSpacing: 2,
          }}>
            Invest Your Time. Compound Your Life.
          </div>
        </div>
      </AbsoluteFill>

      {/* Summary cards — 4 key takeaways */}
      <AbsoluteFill style={{ display: 'flex', flexDirection: 'row', alignItems: 'flex-end', justifyContent: 'center', paddingBottom: 180, gap: 24 }}>
        {[
          { icon: '⏰', text: 'Time = Capital', color: '#e9d700' },
          { icon: '📈', text: 'Consistency Compounds', color: '#FFC107' },
          { icon: '🎯', text: 'Allocate Intentionally', color: '#F5F5F5' },
          { icon: '🚀', text: 'Start Tonight', color: '#ff0000' },
        ].map((card, i) => {
          const cardSprings = [card1, card2, card3, card4][i];
          return (
            <div key={i} style={{
              transform: `translateY(${interpolate(cardSprings, [0, 1], [60, 0])}px) scale(${cardSprings})`,

              <div style={{ fontSize: 44 }}>{card.icon}</div>

                fontFamily: 'Inter',

      <div style={{
        position: 'absolute', bottom: 60, left: '50%',
        transform: `translateX(-50%) scale(${subPulse})`,
        opacity: interpolate(frame, [230, 260], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }),
        backgroundColor: '#e9d700',
        borderRadius: 60,
        paddingLeft: 60, paddingRight: 60,
        paddingTop: 20, paddingBottom: 20,
        boxShadow: '0 0 40px rgba(233,215,0,0.3)',

          fontFamily: 'Inter',
          fontSize: 36,

          color: '#1C1C1C',
          letterSpacing: 2,
          whiteSpace: 'nowrap',

          🔔 Channel Subscribe करो — ProBuffet HQ
        </div>

      {/* Floating particles — grand */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ position: 'absolute', pointerEvents: 'none' }}>
          {[...Array(50)].map((_, i) => {
            const drift = Math.sin(frame * 0.02 + i * 0.8) * 15;
            const floatY = (seededRandom(i * 6) * 1080 - (frame * seededRandom(i * 6 + 1) * 0.4)) % 1080;

              <circle key={i}

            {/* Both arms UP — celebrating */}
            <line x1="0" y1="-24" x2={-65} y2={-75 + Math.sin(frame * 0.1) * 8} stroke="#F5F5F5" strokeWidth="3" />
            <line x1="0" y1="-24" x2={65} y2={-75 + Math.sin(frame * 0.1) * 8} stroke="#F5F5F5" strokeWidth="3" />
            <line x1="0" y1="26" x2="-26" y2="82" stroke="#F5F5F5" strokeWidth="3.5" />
            <line x1="0" y1="26" x2="26" y2="82" stroke="#F5F5F5" strokeWidth="3.5" />
          </g>

          {/* Right figure */}
          <circle cx="1720" cy="900" r="150" fill="url(#glowFinal)" opacity="0.5" />
          <g transform={`translate(1720, 860) rotate(${Math.sin(frame * 0.04 + 1) * 3})`}
            style={{ transformOrigin: '0px 0px' }}
            opacity={sp(50)}>
            <circle cx="0" cy={-90 + Math.sin(frame * 0.05 + 1) * 2} r="38" fill="none" stroke="#F5F5F5" strokeWidth="3.5" />
            <ellipse cx="-12" cy={-94 + Math.sin(frame * 0.05 + 1) * 2} rx="10" ry="8" fill="none" stroke="#e9d700" strokeWidth="2.5" />
            <ellipse cx="12" cy={-94 + Math.sin(frame * 0.05 + 1) * 2} rx="10" ry="8" fill="none" stroke="#e9d700" strokeWidth="2.5" />
            <line x1="-2" y1={-94 + Math.sin(frame * 0.05 + 1) * 2} x2="2" y2={-94 + Math.sin(frame * 0.05 + 1) * 2} stroke="#e9d700" strokeWidth="2" />
            <path d={`M-14,${-72 + Math.sin(frame * 0.05 + 1) * 2} Q0,${-58 + Math.sin(frame * 0.05 + 1) * 2} 14,${-72 + Math.sin(frame * 0.05 + 1) * 2}`} fill="none" stroke="#e9d700" strokeWidth="3" />
            <line x1="0" y1="-52" x2="0" y2="26" stroke="#F5F5F5" strokeWidth="3.5" />
            <line x1="0" y1="-24" x2={-65} y2={-75 + Math.sin(frame * 0.1 + 1) * 8} stroke="#F5F5F5" strokeWidth="3" />
            <line x1="0" y1="-24" x2={65} y2={-75 + Math.sin(frame * 0.1 + 1) * 8} stroke="#F5F5F5" strokeWidth="3" />
            <line x1="0" y1="26" x2="-26" y2="82" stroke="#F5F5F5" strokeWidth="3.5" />
            <line x1="0" y1="26" x2="26" y2="82" stroke="#F5F5F5" strokeWidth="3.5" />
          </g>
        </svg>
      </AbsoluteFill>

      {/* Vignette */}
      <AbsoluteFill style={{
        background: 'radial-gradient(ellipse at center, transparent 35%, rgba(0,0,0,0.7) 100%)',
        pointerEvents: 'none',
      }} />

      {/* Watermark — bigger on outro */}
      <div style={{
        position: 'absolute', bottom: 30, right: 45,
        fontFamily: 'Inter', fontSize: 22, fontWeight: 600,
        color: '#9E9E9E', opacity: 0.4,
      }}>
        ProBuffet HQ
      </div>
    </AbsoluteFill>
  );
};
