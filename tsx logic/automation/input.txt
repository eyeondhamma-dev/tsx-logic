  const rightSlide = interpolate(frame, [30, 60], [500, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const leftOp = sp(30);
  const rightOp = sp(40);

  const pulse = Math.sin(frame * 0.08) * 3;

  const qOp = sp(75);

        {/* Animated shape layers */}
        {[...Array(6)].map((_, i) => {

          const size = 2 + seed * 3;
          return (
            <div key={`p-${i}`} style={{
              position: 'absolute',
              left: x, top: y,
              width: size, height: size,
              borderRadius: '50%',
